9 /**
10   * Validations are an example of [[∪]] implementations offering a left injection to be a [[scalaz.Semigroup]].
11   * [[∪]] will be attached an [[scalaz.Monad]]/[[scalaz.Applicative]] typeclass cumulating the [[scalaz.Semigroup]]
12   * behaviour on the [[scalaz.Applicative]] strategy application.
13   *
14   * @example
15   * {{{
16   *   (fail[Int](DumbError("fail!")) ⊛ fail[Int](DumbError("fail!"))) { _ + _ }
17   *   //will result in
18   *   fail(DumbError("fail!"), DumbError("fail!"))
19   * }}}
20   *
21   *
22   * We define a [[Safely]] safe application build as a [[Kleisli]] representing a function `A => Safe[B]`
23   * [[Safely]] usage of [[scalaz.Applicative]] will exhibit the same behaviour as its [[Safe]] counterpart
24   *
25   * @example
26   * {{{
27   *   alwaysFail[Any, Int](DumbError("fail!")) ⊛ alwaysFail[Any, Int](DumbError("fail!"))) { _ + _ }
28   *   //results in
29   *   fail(DumbError("fail!"), DumbError("fail!"))
30   * }}}
31   *
32   */
33 private[fp] trait Validations {
34   type Safe[A] = NonEmptyList[AppError] ∪ A
35 
36   final def safe[A: ClassTag](a: ⇒ A): Safe[A] = Try(a) match {
37     case scala.util.Success(value) ⇒ injectR(value)
38     case scala.util.Failure(ex)    ⇒ fail(ThrownError(ex.getMessage))
39   }
40 
41   final def fail[A: ClassTag](a: AppError, as: AppError*): Safe[A] = injectL(NonEmptyList(a, as:_*))
42 
43   type Safely[A, B] = Kleisli[Safe, A, B]
44 
45   final def safely[A, B: ClassTag](f: A ⇒ B): Safely[A, B] = Kleisli { a ⇒ safe(f(a)) }
46 
47   implicit def safelyOps[A, B](s: Safely[A, B]): SafelyOps[A, B] = new SafelyOps[A, B](s)
48 
49   implicit def fishOps[A, B](f: A ⇒ Safe[B]): FishOps[A, B] = new FishOps[A, B](f)
50 
51   implicit def safeOps[A: ClassTag](s: Safe[A]): SafeOps[A] = new SafeOps(s)
52 }

61 final class SafelyOps[A, B](val s: Safely[A, B]) extends AnyVal {
62   /**
63     * {{{
64     *    (safely[Any, Int](_ ⇒ 12) ||| safely[Any, Int](_ ⇒ 17))
65     *      // gives safe(12)
66     *
67     *    (alwaysFail[Any, Int](DumbError("fail!")) ||| safely[Any, Int](_ ⇒ 17))
68     *      // gives safe(17)
69     *
70     *    (alwaysFail[Any, Int](DumbError("fail!")) ||| alwaysFail[Any, Int](DumbError("fail!")))
71     *    //gives fail(DumbError("fail!"), DumbError("fail!"))
72     * }}}
73     */
74   def |||(other: ⇒ Safely[A, B]): Safely[A, B] = Kleisli[Safe, A, B] { a ⇒
75     s(a) match {
76       case \/-(b) ⇒ \/-(b)
77       case -\/(e) ⇒ other.run(a) match {
78         case \/-(b) ⇒ \/-(b)
79         case -\/(ee)  ⇒ -\/(e.append(ee))
80       }
81     }
82   }
83 }
