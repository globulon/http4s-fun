12 private[fp] trait Disjunctions {
13   type ∪[A, B] = \/[A, B]
14 
15   final def injectR[A, B](b: B): A ∪ B = \/-(b)
16   final def injectL[A, B](a: A): A ∪ B = -\/(a)
17 
18   implicit def safeMonad[E : Semigroup]: Monad[E ∪ ?] = new Monad[E ∪ ?] {
19     override def point[A](a: ⇒ A): E ∪ A = injectR(a)
20 
21     override def ap[A, B](fa: ⇒ E ∪ A)(f: ⇒ E ∪ (A ⇒ B)): E ∪ B = (fa, f) match {
22       case (\/-(a), \/-(app)) ⇒ injectR(app(a))
23       case (-\/(a), -\/(b))   ⇒ injectL(implicitly[Semigroup[E]].append(a, b))
24       case (-\/(a), _)        ⇒ injectL(a)
25       case (_, -\/(b))        ⇒ injectL(b)
26     }
27 
28     override def bind[A, B](fa: E ∪ A)(f: A ⇒ E ∪ B): E ∪ B = fa match {
29       case \/-(a) ⇒ f(a)
30       case -\/(b) ⇒ -\/(b)
31     }
32   }
