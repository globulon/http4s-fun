8 /**
9   * Defines a generic DSL to extract information from a configuration abstraction.
10   * Information is retrieved by key value
11   *
12   * The DSL exposes 3 methods to resolve configuration values by type
13   *
14   * {{{
15   *   //returns a safe/unsafe action which results will depend on the existence
16   *   //of a configuration element of Type T matching the key entry
17   *   resolve[T](key)
18   *
19   *   //returns an always safe action returning a value of type T matching the key entry
20   *   //or the provided default value of type T when the key entry is missing
21   *   resolve[T](key, orElse = default)
22   *
23   *   //returns a safe action providing a value of type M[T] given by the natural
24   *   //transformation between the Functor [[Safe]] and the Functor [[M[_]]]
25   *   resolve[M[_], T](key)
26   * }}}
27   *
28   * Application of the resolve methods to find a value of type T will rely on the existence of an [[ReadConfig]]
29   * Typeclass for the type T representing the possibility of applying a  function `Config ⇒ String ⇒ T`
30   *
31   * Any DSL implementation will have to define the Config type instance that is being read and provide a set
32   * of implicit [[ReadConfig]] typeclasses
33   *
34   * We provide a default implementation for <a href="https://github.com/lightbend/config">Lightbend Config</a>
35   * classic library in the package 'com.graboids.config.hocon'
36   */
37 trait Configuration {
38   type Config
39   type ResolveConfig[A] = Safely[Config, A]
40 
41   final def resolve[T: ClassTag](k: String)(implicit R: ReadConfig[Config, T]): ResolveConfig[T] =
42     safely[Config, T](R(_)(k))
43 
44   final def resolve[T: ClassTag](k: String, orElse: ⇒ T)(implicit R: ReadConfig[Config, T]): ResolveConfig[T] =
45     safely[Config, T](R(_)(k)) ||| safely(_ ⇒ orElse)
46 
47   final def resolve[M[_], T: ClassTag](k: String)
48                                       (implicit R: ReadConfig[Config, T],
49                                        N: Safe ~> M,
50                                        C: ClassTag[M[T]]): ResolveConfig[M[T]] =
51     safely(resolve[T](k).mapK[M, T](implicitly[Safe ~> M]))
52 
53 
54   def fromString: Safely[String, Config]
55   def fromResource: Safely[String, Config]
56   def fromEnv: Safe[Config]
